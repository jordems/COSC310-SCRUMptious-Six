CREATE TABLE User(
	uid int NOT NULL AUTO_INCREMENT,
	username varchar(30) NOT NULL UNIQUE,
	password char(128) NOT NULL,
	email varchar(50) NOT NULL UNIQUE,
	firstName varchar(30) NOT NULL,
	lastName varchar(30) NOT NULL,
	billingAddress varchar(255) NOT NULL,
	PRIMARY KEY (uid)
) ENGINE = InnoDB;

CREATE TABLE Login_attempts(
	compid int(11) NOT NULL,
	time varchar(30) NOT NULL,
	PRIMARY KEY (compid)
) ENGINE = InnoDB;

CREATE TABLE Forgot(
    uid int NOT NULL,
    code CHAR(10) NOT NULL,
    timeStamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    browserString CHAR(10),
    PRIMARY KEY(uid, code),
    FOREIGN KEY (uid) REFERENCES Users(uid)
)ENGINE = InnoDB;

CREATE TABLE Account(
    aid int NOT NULL,
    uid int NOT NULL,
    balance DECIMAL(20,2) NOT NULL,
    financialinstitution VARCHAR(255) NOT NULL,
    title VARCHAR(255) NOT NULL,
    type VARCHAR(255) NOT NULL,
    PRIMARY KEY(uid, aid),
    FOREIGN KEY (uid) REFERENCES Users(uid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION
)ENGINE = InnoDB;

CREATE TABLE AccountTransaction(
    tid CHAR(64) NOT NULL,
	uid int NOT NULL,
	aid int NOT NULL, 
    date DATE NOT NULL,
    statementName VARCHAR(255) NOT NULL,
    amount DECIMAL(18,2) NOT NULL,
    `desc` VARCHAR(255),
    PRIMARY KEY(tid),
    FOREIGN KEY (uid) REFERENCES Users(uid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION,
    FOREIGN KEY (aid) REFERENCES Account(aid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION
)ENGINE = InnoDB;

CREATE TABLE Wallet(
    wid int NOT NULL,
    balance DECIMAL(20,2),
    isFrozen boolean NOT NULL DEFAULT 0,
    PRIMARY KEY (wid),
    FOREIGN KEY (wid) REFERENCES Users(uid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION
) ENGINE = InnoDB;

CREATE TABLE Transaction(
    tid CHAR(64) NOT NULL, 
    fromid int NOT NULL,
    toid int NOT NULL,
    datetime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    amount DECIMAL(18,2) NOT NULL,
    PRIMARY KEY (tid),
    FOREIGN KEY (fromid) REFERENCES Wallet(wid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION,
    FOREIGN KEY (toid) REFERENCES Wallet(wid)
    	ON UPDATE NO ACTION ON DELETE NO ACTION
) ENGINE = InnoDB;
